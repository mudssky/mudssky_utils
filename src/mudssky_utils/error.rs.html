7/// Argument error for invalid function arguments
8#[derive(Error, Debug, Clone)]
9#[error("Argument error: {message}")]
10pub struct ArgumentError {
11    pub message: String,
12}
13
14impl ArgumentError {
15    /// Create a new ArgumentError
16    pub fn new(message: impl Into<String>) -> Self {
17        Self {
18            message: message.into(),
19        }
20    }
21}
22
23/// Validation error for data validation failures
24#[derive(Error, Debug, Clone)]
25#[error("Validation error for field '{field}': {message}{}", value.as_ref().map(|v| format!(" (value: '{v}')")).unwrap_or_default())]
26pub struct ValidationError {
27    field: String,
28    message: String,
29    value: Option<String>,
30}
31
32impl ValidationError {
33    /// Create a new ValidationError
34    pub fn new(field: impl Into<String>, message: impl Into<String>) -> Self {
35        Self {
36            field: field.into(),
37            message: message.into(),
38            value: None,
39        }
40    }
41
42    /// Create a new ValidationError with the invalid value
43    pub fn with_value(
44        field: impl Into<String>,
45        message: impl Into<String>,
46        value: impl Into<String>,
47    ) -> Self {
48        Self {
49            field: field.into(),
50            message: message.into(),
51            value: Some(value.into()),
52        }
53    }
54
55    /// Get the field name
56    pub fn field(&self) -> &str {
57        &self.field
58    }
59
60    /// Get the error message
61    pub fn message(&self) -> &str {
62        &self.message
63    }
64
65    /// Get the invalid value if available
66    pub fn value(&self) -> Option<&str> {
67        self.value.as_deref()
68    }
69}

195/// Generic utility error that can wrap other errors
196#[derive(Error, Debug)]
197pub enum UtilsError {
198    #[error(transparent)]
199    Argument(#[from] ArgumentError),
200
201    #[error(transparent)]
202    Validation(#[from] ValidationError),
203
204    #[error(transparent)]
205    Config(#[from] ConfigError),
206
207    #[error(transparent)]
208    Network(#[from] NetworkError),
209
210    #[error(transparent)]
211    Parse(#[from] ParseError),
212
213    #[error("IO error: {0}")]
214    Io(#[from] std::io::Error),
215
216    #[error("Error: {0}")]
217    Other(#[from] Box<dyn std::error::Error + Send + Sync>),
218}
219
220/// Result type alias for utils operations
221pub type UtilsResult<T> = Result<T, UtilsError>;
222
223/// Create an argument error
224pub fn argument_error(message: impl Into<String>) -> ArgumentError {

238/// Create a network error

240    NetworkError::new(operation, message)

243/// Create a parse error
